#include <iostream>
using namespace std;
char s[100];
long int nr,suma = 0;
int main()
{
    
    
    FILE* sir; 
    fopen_s(&sir, "in.txt", "r");
    while (fgets(s, 100, sir))
    {
        for (int i = 0; s[i] != NULL; i++)
        {
            if (s[i] >= '0' && s[i] <= '9')
                nr = nr * 10 + (s[i] - '0');
            else
            {
                suma = suma + nr;
                nr = 0;
            }
        }
        suma = suma + nr;
    }
    printf("%d",suma);
    return 0;
}
