#include <iostream>
using namespace std;

ifstream fin("in.txt");
int main()
{
    char s[100];
    int nr = 0,suma=0;
    FILE* sir = fopen("in.txt", "r");
    while(fgets(s,100,sir))
    {for (int i = 0; s[i] != NULL; i++)
    {
        if (s[i] >= '0' && s[i] <= '9')
            nr = nr * 10 + (s[i] - '0');
        else
        {
            suma = suma + nr;
            nr = 0;
        }
    }
    printf("%d",suma);
    return 0;
}
